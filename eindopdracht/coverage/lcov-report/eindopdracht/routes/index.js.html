var isAuth = function(req, res, next)
{
	if(req.isAuthenticated())
		return next();
	res.redirect('/login');
}
var isAdmin = function(req, res, next)
{
	if(req.isAuthenticated() && req.user.admin)
		return next();
	res.redirect('/login');
}
